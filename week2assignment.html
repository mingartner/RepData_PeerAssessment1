df1 <- activitydata %>% group_by(date) %>% summarize (

     xlab = "Daily Steps", ylab = "Frequency (days)", xlim = c(-100, 25000),

mean_text <- paste0("mean", " (", round(mean(df1$dailytotal), 1), ")")
